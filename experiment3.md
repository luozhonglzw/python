**提示：

代码提交地址
<https://www.codewars.com/kata/5277c8a221e209d3f6000b56>


```python
def valid_braces(string):
  result = []
  for i in string:
      if i in "({[":
          result.append(i)
      elif len(result) != 0:
          if (i == ')') & (result[-1] == '('):
              result.pop()
          elif (i == '}') & (result[-1] == '{'):
              result.pop()
          elif result[-1] == '[':
              result.pop()
  if len(result) == 0:
      return True
  else:
      return False

```

def recoverSecret(triplets):
    'triplets is a list of triplets from the secrent string. Return the string.'
    lic = []
    for i in range(len(triplets)):
        for j in range(len(triplets[i])):
            lic.append(triplets[i][j])
 
    lic = set(lic)
    lic = list(lic)
    while True:
        cnt = 0
        for i in range(len(triplets)):
            for j in range(len(triplets[i]) - 1):
                a = triplets[i][j]
                b = triplets[i][j + 1]
                index_a = lic.index(a)
                index_b = lic.index(b)
                if index_a > index_b:
                    temp = lic[index_b]
                    lic[index_b] = lic[index_a]
                    lic[index_a] = temp
                    cnt += 1
        if cnt == 0:
            return ''.join(lic)
